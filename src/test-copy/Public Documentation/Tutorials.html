DataModel (data, schema, options)
parameters:
	- data: 
			required: true
			type: Array, CSV, JSON
			descriptions: Data to be fed to datamodel
	- schema:
			required: true
			type: JSON
			description : The field definitions in json format.
	- options:
			required: false
			type: Object
			default: {}
			description: Additional config
			parameters:
					- name:
							required: false
							type: String
							default: ''
							descriptions: Alias for datamodel
					- fieldSepeartor:
							required: false
							type: String
							default: ','
							descriptions: Used only for CSV string data.
